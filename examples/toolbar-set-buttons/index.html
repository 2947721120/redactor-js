Change Available Buttons 


<script>
$(function()
{
    // By default 
    // ['format', 'bold', 'italic', 'deleted', 'lists', 'image', 'file', 'link', 'horizontalrule']
    // Change it:
    
    $('#redactor').redactor({
        focus: true,
        buttons: ['format', 'bold', 'italic', 'horizontalrule']
    });
});
</script>
