动态文本 


<script>
$(function()
{
    var textarea = $('<textarea style="height: 200px;">');
    $('#testdynamic').append(textarea);
    $(textarea).redactor({
        focus: true,
        callbacks: {
            init: function()
            {
                this.code.set("<p>Lorem ipsum...</p>");
            }
        }
    });
});
</script>
 
<div id="testdynamic"></div>
